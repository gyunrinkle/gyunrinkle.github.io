# 계수 정렬
---

```python
import sys

# sys.stdin = open("input.txt", "r")

input = sys.stdin.readline

N, M = None, None
상근_카드 = [False for _ in range(2 * (10**7) + 1)]
숫자들 = None


def get_상근_카드_index(i):
    return i + 10**7


def read_input():
    global N, M, 상근_카드, 숫자들
    N = int(input().rstrip())
    for 카드 in input().rstrip().split():
        상근_카드[get_상근_카드_index(int(카드))] = True

    M = int(input().rstrip())
    숫자들 = [*map(int, input().rstrip().split())]


def solve():

    for 숫자 in 숫자들:
        if 상근_카드[get_상근_카드_index(숫자)]:
            print("1", end=" ")
        else:
            print("0", end=" ")


if __name__ == "__main__":
    read_input()
    solve()

```

# 이분 탐색 `s < e`

```python
import sys

# sys.stdin = open("input.txt", "r")
input = sys.stdin.readline


def read_input():
    global N, K
    global 상근_카드, 정수들

    N = int(input().rstrip())
    상근_카드 = sorted(map(int, input().rstrip().split()))
    M = int(input().rstrip())
    정수들 = [*map(int, input().rstrip().split())]


def binary_search(x):
    s = 0
    e = len(상근_카드)
    while s < e:
        m = (s + e) // 2
        if 상근_카드[m] == x:
            return True

        if x < 상근_카드[m]:
            e = m
        else:
            s = m + 1

    return False


def solve():
    for 정수 in 정수들:
        print(f"{1 if binary_search(정수) else 0}", end=" ")


if __name__ == "__main__":
    read_input()

    solve()

```

import sys

def read_input():
    global N, K

def binary_search(x):
    s = 0

def solve():
